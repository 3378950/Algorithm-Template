```c++
#include <iostream>
#include <cstring>

using namespace std;

const int N = 1010;
int a[N], dp[N], tr[N];

int main() {
    int n; scanf("%d", &n);
    for(int i = 1; i <= n; i++) scanf("%d", &a[i]);
    
    memset(tr, -1, sizeof tr);
    
    for(int i = 1; i <= n; i++) {
        dp[i] = 1;
        for(int j = 1; j < i; j++) {
            if(a[i] > a[j]) {
                if(dp[i] < dp[j] + 1) {
                    dp[i] = dp[j] + 1;
                    tr[i] = j;
                }
            }
        }
    }
    
    int ans = 0, sign = -1;
    for(int i = 1; i <= n; i++) {
        if(ans < dp[i]) {
            ans = dp[i];
            sign = i;
        }
    }
    printf("%d %d\n", ans, sign);
    //printf("%d ", a[sign]);
    int i = sign;
    while(~i) {
        cout << a[i] << " ";
        i = tr[i];
    }
    return 0;
}
```
