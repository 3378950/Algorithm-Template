## 可用于反映集合之间所属关系的数据结构



### 代码模板 
```c++
class UnionFind {
public:
	const int N = 1e6 + 10;
	int fa[N];
	
	UnionFind(int num) {
		for(int i = 1; i <= num; i++) fa[i] = i;
	}
	
	int Find(int x) {
		return fa[x] = (fa[x] == x) ? x: Find(fa[x]);
	}
	
	void Union(int a, int b) {
		a = Find(a), b = Find(b);
		fa[a] = b;
	}
	
	int Count(int n) {
	    int ans = 0;
	    for(int i = 1; i <= n; i++)
	        if(fa[i] == i) ans ++;
	    return ans;
	}
};
```
